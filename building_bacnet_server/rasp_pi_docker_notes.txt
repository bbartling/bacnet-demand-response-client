# build docker file with:
docker build -t bacnet_dr_server .

# run docker file:
docker run --name bacnet_dr_server -d -p 47808:47808/udp bacnet_dr_server
# docker run -p 8080:8080 bacnet_dr_server

# stop docker container:
docker stop bacnet_dr_server

# remove docker container if you want to make change to .py file rebuild and run again:
docker rm bacnet_dr_server

# tail logs:
docker logs -f bacnet_dr_server

# use with no Docker
pip install bacpypes aiohttp

# run script
python bacnet_server.py
